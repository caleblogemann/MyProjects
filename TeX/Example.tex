    % example table using booktabs package
    \begin{table}
        \centering
        \begin{tabular}{llr}
            % top horizontal line - thicker rule
            \toprule

            % content spanning multiple columns
            \multicolumn{2}{c}{Item} \\

            % horizontal line - spanning only 2 columns
            \cmidrule(r){1-2}

            Animal    & Description & Price (\$) \\
            % horizontal line across entire table - thinner rule
            \midrule
            Gnat      & per gram    & 13.65      \\
                      & each        & 0.01       \\
            Gnu       & stuffed     & 92.50      \\
            Emu       & stuffed     & 33.33      \\
            Armadillo & frozen      & 8.99       \\

            % bottom horizontal line - thicker rule
            \bottomrule
        \end{tabular}

        % include caption
        \caption{Booktabs Example Table}

        % include label - to be referenced elsewhere
        % practice begin all table labels with tab:
        \label{tab:exampleTable}
    \end{table}
